做过这个题了，比较清晰吧就不多说了。这个算法的时间复杂度是O（N），空间复杂度是O（1）。

{% highlight python %}

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, l1: ListNode, l2: ListNode) -> ListNode:
        dummy = new = ListNode(None)
        while l1 or l2:
            if l1 and l2:
                if l1.val <= l2.val:
                    new.next = l1
                    l1 = l1.next
                else:
                    new.next = l2
                    l2 = l2.next
            elif l1 and not l2:
                new.next = l1
                l1 = l1.next
            elif not l1 and l2:
                new.next = l2
                l2 = l2.next
            new = new.next
        return dummy.next
