这题比较简单，遍历两次即可，这个算法的时间复杂度是O（N），空间复杂度是O（1）。

{% highlight python %}

from collections import defaultdict

class Solution:
    def firstUniqChar(self, s: str) -> str:
        hashtable = defaultdict(int)
        for x in s:
            hashtable[x] += 1
        for x in s:
            if hashtable[x] == 1:
                return x
        return " "
