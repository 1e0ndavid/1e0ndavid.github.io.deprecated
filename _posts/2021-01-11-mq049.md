class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        if not digits:
            return []

        d = {
            "2":"abc",
            "3":"def",
            "4":"ghi",
            "5":"jkl",
            "6":"mno",
            "7":"pqrs",
            "8":"tuv",
            "9":"wxyz"
            }

        def backtrack(x):
            if x == len(digits):
                res.append("".join(combination))
            else:
                digit = digits[x]
                for c in d[digit]:
                    combination.append(c)
                    backtrack(x+1)
                    combination.pop()
    
        res = []
        combination = []
        backtrack(0)
        return res
